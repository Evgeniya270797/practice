private static boolean checkHorizontal(char sign) {
        boolean result = false;
        for (int i = 0; i < chars.length; i++) {
            if(chars[i][0]==sign&&chars[i][0]==chars[i][1]&&chars[i][1]==chars[i][2]){
                result=true;
                break;
            }
        }
        return result;
    }
